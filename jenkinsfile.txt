pipeline {
    agent any

    environment {
        VENV = '.venv'}
    stages {
        stage('Clone Repository') {
            steps {
                git branch: 'dev', url: 'https://github.com/faith-nte/capstone'
            }
        }
        stage('Build Addressbook BaseOS Image') {
            steps {
                sh 'docker build â€“t addressbook . || true'
            }
        }

        stage('Launch Addressbook Base OS Container') {
            steps {
                sh 'docker run -p 80:5000 addressbook'
            }
        }

        stage('Install mysql-client') {
            steps {
                sh 'docker exec addressbook apt-get update && docker exec addressbook apt install mysql-server -y'
            }
        }

        stage('Set MySQL Root Password') {
            steps {
                sh '''
                docker exec addressbook service mysql start
                docker exec addressbook sudo mysql <<EOF
                ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY 'yourpassword';
                FLUSH PRIVILEGES;
                EOF
                '''}
        }

        stage('Create SQL Database') {
            steps {
                sh 'docker exec addressbook mysql -u root -p < schema.sql'
            }
        }

        stage('Run Tests') {
            steps {
                script {
                    sh 'docker exec addressbook source venv/bin/activate || true'
                    sh 'docker exec addressbook pytest || true'
                }
            }
        }

        stage('Run Python App') {
            steps {
                script {
                    sh 'docker exec addressbook python run.py'
                }
            }
        }

    post {
        success {
            script {
                sh 'echo "Pipeline executed successfully"'
            }
        }

        failure {
            script {
                sh 'echo "Pipeline failed"'
            }
        }
    }
}
